function bjStartRound() { G.round++; document.getElementById('round-num').textContent = G.round; G.deck = newDeck(); G.bjDealer.hand = []; // Reset each player for the new round G.bjPlayers.forEach(p => { p.bjBust = p.balance <= 0; p.hand = []; p.bjStatus = ''; p.bjResult = ''; p.bjBet = 0; }); // Assign bets and deduct from balance immediately G.bjPlayers.forEach(p => { if (p.bjBust) return; if (p.isHuman) { p.bjBet = Math.min(G.bjPlayerBet || 10, p.balance); } else { p.bjBet = Math.max(10, Math.floor(p.balance * (0.1 + Math.random() * 0.2))); p.bjBet = Math.min(p.bjBet, p.balance); } p.balance -= p.bjBet; // pre-deduct bet }); // Deal 2 cards to each active player, then 2 to dealer const active = G.bjPlayers.filter(p => !p.bjBust); active.forEach(p => p.hand.push(G.deck.pop())); G.bjDealer.hand.push(G.deck.pop()); active.forEach(p => p.hand.push(G.deck.pop())); G.bjDealer.hand.push(G.deck.pop()); log(`--- Blackjack Round ${G.round} ---`, 'log-info'); // Turn order: bots first, human last G.bjTurnOrder = [ ...G.bjPlayers.filter(p => !p.isHuman && !p.bjBust), ...G.bjPlayers.filter(p => p.isHuman && !p.bjBust) ]; G.bjTurnIndex = 0; G.bjPhase = 'player_turns'; renderBJ(); bjAdvanceTurn(); } function bjPlayerAction(action) { if (G.bjPhase !== 'player_turns') return; const p = G.player; if (p.bjStatus !== '') return; // already acted if (action === 'hit') { p.hand.push(G.deck.pop()); const val = bjValue(p.hand); log(`You hit → ${val}`, 'log-action'); if (val > 21) { p.bjStatus = 'bust'; log('You bust!', 'log-lose'); G.bjTurnIndex++; renderBJ(false); setTimeout(() => bjAdvanceTurn(), 700); } else if (val === 21) { p.bjStatus = 'stand'; G.bjTurnIndex++; renderBJ(false); setTimeout(() => bjAdvanceTurn(), 700); } else { renderBJ(true); // still human's turn, show buttons again } } else if (action === 'stand') { p.bjStatus = 'stand'; log(`You stand at ${bjValue(p.hand)}`, 'log-action'); G.bjTurnIndex++; renderBJ(false); setTimeout(() => bjAdvanceTurn(), 500); } else if (action === 'double') { const extra = Math.min(p.bjBet, p.balance); p.balance -= extra; // deduct extra bet immediately p.bjBet += extra; p.hand.push(G.deck.pop()); const val = bjValue(p.hand); log(`You double down → ${val} (total bet: $${p.bjBet})`, 'log-action'); p.bjStatus = val > 21 ? 'bust' : 'stand'; if (val > 21) log('You bust!', 'log-lose'); G.bjTurnIndex++; renderBJ(false); setTimeout(() => bjAdvanceTurn(), 700); } } function bjResolve() { G.bjPhase = 'result'; const dVal = bjValue(G.bjDealer.hand); const dBust = dVal > 21; const dBJ = bjIsBlackjack(G.bjDealer.hand); log(`Dealer: ${dVal}${dBust ? ' (BUST)' : ''}`, 'log-info'); G.bjPlayers.forEach(p => { if (p.bjBust || !p.bjBet) return; const pVal = bjValue(p.hand); const pBJ = bjIsBlackjack(p.hand); const playerBust = p.bjStatus === 'bust'; if (playerBust) { // Bet already deducted — nothing to return p.bjResult = 'lose'; log(`${p.name} busted. -$${p.bjBet}`, 'log-lose'); } else if (pBJ && dBJ) { // Both blackjack — push: return bet p.balance += p.bjBet; p.bjResult = 'push'; log(`${p.name} push (both blackjack).`, 'log-action'); } else if (pBJ) { // Blackjack pays 3:2 — return bet + 1.5x profit const winnings = p.bjBet + Math.floor(p.bjBet * 1.5); p.balance += winnings; p.bjResult = 'blackjack'; log(`${p.name} BLACKJACK! +$${Math.floor(p.bjBet * 1.5)}`, 'log-win'); } else if (dBust || pVal > dVal) { // Win — return bet + equal profit p.balance += p.bjBet * 2; p.bjResult = 'win'; log(`${p.name} wins! +$${p.bjBet}`, 'log-win'); } else if (pVal === dVal) { // Push — return bet p.balance += p.bjBet; p.bjResult = 'push'; log(`${p.name} push.`, 'log-action'); } else { // Dealer wins — bet already deducted p.bjResult = 'lose'; log(`${p.name} loses. -$${p.bjBet}`, 'log-lose'); } }); renderBJ(false); setTimeout(() => showBJResult(), 900); }
